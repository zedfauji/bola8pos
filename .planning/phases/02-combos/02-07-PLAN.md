```typescript
import fc from 'fast-check';
import { computePoolSessionBilling } from '@shared/lib/pool-billing';
import { describe, it, expect } from 'vitest';

// Pure function under test — defined here, not imported from production code
function computeComboPrice(
  comboPriceOverride: number | null,
  children: { basePrice: number; qty: number }[]
): number {
  if (comboPriceOverride != null) return comboPriceOverride;
  return children.reduce((sum, c) => sum + c.basePrice * c.qty, 0);
}

describe('P2: combo pricing properties', () => {
  it('P2a: when comboPriceOverride is set, it is always used as total', () => {
    fc.assert(
      fc.property(
        fc.float({ min: 0, max: 10000, noNaN: true }),
        fc.array(fc.record({ basePrice: fc.float({ min: 0, max: 1000, noNaN: true }), qty: fc.integer({ min: 1, max: 10 }) })),
        (override, children) => {
          const result = computeComboPrice(override, children);
          expect(result).toBe(override);
        }
      ),
      { numRuns: 500 }
    );
  });

  it('P2b: when comboPriceOverride is null, total = sum(basePrice × qty)', () => {
    fc.assert(
      fc.property(
        fc.array(fc.record({ basePrice: fc.float({ min: 0, max: 1000, noNaN: true }), qty: fc.integer({ min: 1, max: 10 }) })),
        (children) => {
          const expected = children.reduce((s, c) => s + c.basePrice * c.qty, 0);
          expect(computeComboPrice(null, children)).toBeCloseTo(expected, 5);
        }
      ),
      { numRuns: 500 }
    );
  });

  it('P2c: computePoolSessionBilling with prepaidMinutes never produces negative charge', () => {
    fc.assert(
      fc.property(
        fc.integer({ min: 0, max: 300 }),  // elapsedMinutes proxy via equal start/end
        fc.integer({ min: 0, max: 120 }),   // prepaidMinutes
        fc.integer({ min: 10, max: 1000 }), // ratePerHour
        (elapsedMins, prepaidMinutes, ratePerHour) => {
          const startedAt = new Date(2026, 0, 1, 12, 0, 0);
          const endTime = new Date(startedAt.getTime() + elapsedMins * 60 * 1000);
          const result = computePoolSessionBilling({ startedAt, endTime, ratePerHour, prepaidMinutes });
          expect(result.totalCharge).toBeGreaterThanOrEqual(0);
        }
      ),
      { numRuns: 500 }
    );
  });

  it('P2d: when prepaid >= elapsed, totalCharge === 0', () => {
    fc.assert(
      fc.property(
        fc.integer({ min: 0, max: 120 }),
        fc.integer({ min: 0, max: 120 }),
        fc.integer({ min: 10, max: 1000 }),
        (elapsedMins, extraPrepaid, ratePerHour) => {
          const prepaidMinutes = elapsedMins + extraPrepaid;
          const startedAt = new Date(2026, 0, 1, 12, 0, 0);
          const endTime = new Date(startedAt.getTime() + elapsedMins * 60 * 1000);
          const result = computePoolSessionBilling({ startedAt, endTime, ratePerHour, prepaidMinutes });
          expect(result.totalCharge).toBe(0);
        }
      ),
      { numRuns: 300 }
    );
  });
});
```

```typescript
/**
 * seed-combos.ts
 *
 * Seeds demo combo products: Cubeta Regular, Cubeta Premium, Martes de Cubeta + Pool.
 * Idempotent — safe to re-run (uses upsert on product name).
 *
 * Usage: cd bar-pos && npx tsx scripts/seed-combos.ts
 * Requires: SUPABASE_URL and SUPABASE_SERVICE_ROLE_KEY in .env.local
 *
 * WARNING: Uses service role key — do NOT import this in the renderer.
 */

import { createClient } from '@supabase/supabase-js';
import * as dotenv from 'dotenv';
import * as path from 'path';
import * as fs from 'fs';

// Load .env.local
const envPath = path.resolve(__dirname, '../.env.local');
if (fs.existsSync(envPath)) {
  dotenv.config({ path: envPath });
}

const SUPABASE_URL = process.env['VITE_SUPABASE_URL'] ?? process.env['SUPABASE_URL'];
const SUPABASE_SERVICE_ROLE_KEY = process.env['SUPABASE_SERVICE_ROLE_KEY'];

if (!SUPABASE_URL || !SUPABASE_SERVICE_ROLE_KEY) {
  console.error('Missing SUPABASE_URL or SUPABASE_SERVICE_ROLE_KEY in .env.local');
  process.exit(1);
}

// eslint-disable-next-line @typescript-eslint/no-explicit-any
const db = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY) as any;

async function main() {
  console.log('Seeding combo products...');

  // 1. Ensure a "Combos" category exists
  const { data: categoryData, error: catErr } = await db
    .from('categories')
    .upsert({ name: 'Combos', color: '#6366f1', sort_order: 99, is_food: false }, { onConflict: 'name' })
    .select('id')
    .single();
  if (catErr) { console.error('Category upsert failed:', catErr); process.exit(1); }
  const comboCategoryId: string = (categoryData as { id: string }).id;

  // 2. Find "Regular" and "Premium" beer categories
  const { data: regularCat } = await db.from('categories').select('id').eq('name', 'Regular').maybeSingle();
  const { data: premiumCat } = await db.from('categories').select('id').eq('name', 'Premium').maybeSingle();
  const regularCategoryId: string | null = (regularCat as { id: string } | null)?.id ?? null;
  const premiumCategoryId: string | null = (premiumCat as { id: string } | null)?.id ?? null;

  // 3. Upsert Cubeta Regular product
  const { data: cubetaRegular, error: crErr } = await db
    .from('products')
    .upsert({
      name: 'Cubeta Regular',
      category_id: comboCategoryId,
      base_price: 350,
      is_active: true,
      is_combo: true,
      combo_eligible: false,  // the combo itself is not eligible as a child
      combo_price_override: 350,
    }, { onConflict: 'name' })
    .select('id')
    .single();
  if (crErr) { console.error('Cubeta Regular upsert failed:', crErr); process.exit(1); }
  const cubetaRegularId: string = (cubetaRegular as { id: string }).id;

  // 4. Upsert slot for Cubeta Regular
  const { data: slotRegular } = await db
    .from('combo_slots')
    .upsert({
      combo_product_id: cubetaRegularId,
      label: 'Beers',
      slot_type: 'product',
      min_qty: 10,
      max_qty: 10,
      is_required: true,
      sort_order: 0,
    }, { onConflict: 'combo_product_id,label' })
    .select('id')
    .single();
  const slotRegularId: string = (slotRegular as { id: string }).id;

  // 5. Populate options from Regular beer category (if exists)
  if (regularCategoryId) {
    const { data: regularBeers } = await db
      .from('products')
      .select('id')
      .eq('category_id', regularCategoryId)
      .eq('combo_eligible', true)
      .eq('is_active', true);
    for (const beer of (regularBeers as { id: string }[] | null) ?? []) {
      await db.from('combo_slot_options').upsert({
        combo_slot_id: slotRegularId,
        child_product_id: beer.id,
        sort_order: 0,
      }, { onConflict: 'combo_slot_id,child_product_id' });
    }
  }

  // 6. Upsert Cubeta Premium
  const { data: cubetaPremium } = await db
    .from('products')
    .upsert({
      name: 'Cubeta Premium',
      category_id: comboCategoryId,
      base_price: 450,
      is_active: true,
      is_combo: true,
      combo_eligible: false,
      combo_price_override: 450,
    }, { onConflict: 'name' })
    .select('id')
    .single();
  const cubetaPremiumId: string = (cubetaPremium as { id: string }).id;

  const { data: slotPremium } = await db
    .from('combo_slots')
    .upsert({
      combo_product_id: cubetaPremiumId,
      label: 'Beers',
      slot_type: 'product',
      min_qty: 10,
      max_qty: 10,
      is_required: true,
      sort_order: 0,
    }, { onConflict: 'combo_product_id,label' })
    .select('id')
    .single();
  const slotPremiumId: string = (slotPremium as { id: string }).id;

  if (premiumCategoryId) {
    const { data: premiumBeers } = await db
      .from('products')
      .select('id')
      .eq('category_id', premiumCategoryId)
      .eq('combo_eligible', true)
      .eq('is_active', true);
    for (const beer of (premiumBeers as { id: string }[] | null) ?? []) {
      await db.from('combo_slot_options').upsert({
        combo_slot_id: slotPremiumId,
        child_product_id: beer.id,
        sort_order: 0,
      }, { onConflict: 'combo_slot_id,child_product_id' });
    }
  }

  // 7. Upsert Martes de Cubeta + Pool (Tuesday only)
  const { data: martesCubeta } = await db
    .from('products')
    .upsert({
      name: 'Martes de Cubeta + Pool',
      category_id: comboCategoryId,
      base_price: 400,
      is_active: true,
      is_combo: true,
      combo_eligible: false,
      combo_price_override: 400,
    }, { onConflict: 'name' })
    .select('id')
    .single();
  const martesCubetaId: string = (martesCubeta as { id: string }).id;

  // Beers slot
  const { data: martesBeersSlot } = await db
    .from('combo_slots')
    .upsert({
      combo_product_id: martesCubetaId,
      label: 'Beers',
      slot_type: 'product',
      min_qty: 6,
      max_qty: 6,
      is_required: true,
      sort_order: 0,
    }, { onConflict: 'combo_product_id,label' })
    .select('id')
    .single();
  const martesBeersSlotId: string = (martesBeersSlot as { id: string }).id;

  if (regularCategoryId) {
    const { data: regularBeers } = await db
      .from('products')
      .select('id')
      .eq('category_id', regularCategoryId)
      .eq('combo_eligible', true)
      .eq('is_active', true);
    for (const beer of (regularBeers as { id: string }[] | null) ?? []) {
      await db.from('combo_slot_options').upsert({
        combo_slot_id: martesBeersSlotId,
        child_product_id: beer.id,
        sort_order: 0,
      }, { onConflict: 'combo_slot_id,child_product_id' });
    }
  }

  // Pool time slot
  const { data: poolSlot } = await db
    .from('combo_slots')
    .upsert({
      combo_product_id: martesCubetaId,
      label: 'Pool Time',
      slot_type: 'pool_time',
      min_qty: 1,
      max_qty: 1,
      is_required: true,
      sort_order: 1,
    }, { onConflict: 'combo_product_id,label' })
    .select('id')
    .single();
  const poolSlotId: string = (poolSlot as { id: string }).id;

  await db.from('combo_slot_options').upsert({
    combo_slot_id: poolSlotId,
    child_product_id: null,
    prepaid_minutes: 60,
    sort_order: 0,
  }, { onConflict: 'combo_slot_id,sort_order' });

  // Availability: Tuesday only (ISO 2)
  await db.from('combo_availability').delete().eq('combo_product_id', martesCubetaId);
  await db.from('combo_availability').insert({
    combo_product_id: martesCubetaId,
    days_of_week: [2],  // Tuesday
    start_time: null,
    end_time: null,
    start_date: null,
    end_date: null,
  });

  console.log('Combo seed complete:');
  console.log('  - Cubeta Regular:', cubetaRegularId);
  console.log('  - Cubeta Premium:', cubetaPremiumId);
  console.log('  - Martes de Cubeta + Pool:', martesCubetaId);
}

main().catch(e => { console.error(e); process.exit(1); });
```

**Note on upsert conflict targets:** The `onConflict` for combo_slots uses `combo_product_id,label` — this requires a unique index in the DB. If that index doesn't exist, change to `.select().eq('combo_product_id', ...).eq('label', ...).maybeSingle()` then insert if not found. Read the migration file before deciding.

**Note on `combo_eligible: false` on the combo products themselves:** The seed sets `combo_eligible=false` on the combo products (Cubeta Regular, etc.) so they cannot be nested as children of another combo. This is consistent with the no-nesting design.
