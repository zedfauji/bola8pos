**Write queries.ts:**

```typescript
/* eslint-disable @typescript-eslint/no-explicit-any, @typescript-eslint/no-unsafe-assignment,
   @typescript-eslint/no-unsafe-call, @typescript-eslint/no-unsafe-member-access,
   @typescript-eslint/no-unsafe-return */
// Pre-regen cast: role_permissions table added in Phase 13; supabase.types.ts extended manually.
import { useQuery } from '@tanstack/react-query';
import { supabase } from '@shared/lib/supabase';
import { logger } from '@shared/lib/logger-instance';
import { err, ok, type Result } from '@shared/lib/result';
import type { StaffAction, StaffRole } from '@shared/lib/rbac';

const db = supabase as any;

export const rbacKeys = {
  all: ['role_permissions'] as const,
  list: () => [...rbacKeys.all, 'list'] as const,
};

export function useRolePermissions() {
  return useQuery({
    queryKey: rbacKeys.list(),
    queryFn: async (): Promise<Result<Map<StaffRole, Set<StaffAction>>>> => {
      const { data, error } = await db.from('role_permissions').select('*');
      if (error) {
        logger.error('useRolePermissions: query failed', { error });
        return err({
          code: 'SUPABASE_ERROR' as const,
          message: (error as { message?: string }).message ?? 'Unknown error',
        });
      }
      const map = new Map<StaffRole, Set<StaffAction>>();
      for (const row of (data ?? []) as Record<string, unknown>[]) {
        const role = row['role'] as StaffRole;
        const action = row['action'] as StaffAction;
        if (!map.has(role)) map.set(role, new Set<StaffAction>());
        map.get(role)!.add(action);
      }
      return ok(map);
    },
    staleTime: 30 * 1000,
  });
}
```

import { describe, it, expect, vi, beforeEach } from 'vitest';
import { renderHook, waitFor } from '@testing-library/react';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import React from 'react';

// Mock supabase
vi.mock('@shared/lib/supabase', () => ({
  supabase: {
    from: vi.fn(),
  },
}));

import { supabase } from '@shared/lib/supabase';
import { useRolePermissions } from './queries';

const createWrapper = () => {
  const queryClient = new QueryClient({
    defaultOptions: { queries: { retry: false } },
  });
  return ({ children }: { children: React.ReactNode }) =>
    React.createElement(QueryClientProvider, { client: queryClient }, children);
};

describe('useRolePermissions', () => {
  beforeEach(() => {
    vi.clearAllMocks();
  });

  it('returns Map with correct role→action entries', async () => {
    const mockData = [
      { id: 'id1', role: 'bartender', action: 'create_order', created_at: '2026-01-01' },
      { id: 'id2', role: 'bartender', action: 'close_tab', created_at: '2026-01-01' },
      { id: 'id3', role: 'admin', action: 'manage_staff', created_at: '2026-01-01' },
    ];
    (supabase as any).from = vi.fn().mockReturnValue({
      select: vi.fn().mockResolvedValue({ data: mockData, error: null }),
    });

    const { result } = renderHook(() => useRolePermissions(), {
      wrapper: createWrapper(),
    });

    await waitFor(() => expect(result.current.isSuccess).toBe(true));

    const queryResult = result.current.data;
    expect(queryResult?.ok).toBe(true);
    if (!queryResult?.ok) return;

    const map = queryResult.data;
    expect(map.get('bartender')?.has('create_order')).toBe(true);
    expect(map.get('bartender')?.has('close_tab')).toBe(true);
    expect(map.get('admin')?.has('manage_staff')).toBe(true);
    expect(map.get('kitchen')).toBeUndefined();
  });

  it('returns ok with empty Map when no rows in DB', async () => {
    (supabase as any).from = vi.fn().mockReturnValue({
      select: vi.fn().mockResolvedValue({ data: [], error: null }),
    });

    const { result } = renderHook(() => useRolePermissions(), {
      wrapper: createWrapper(),
    });

    await waitFor(() => expect(result.current.isSuccess).toBe(true));

    const queryResult = result.current.data;
    expect(queryResult?.ok).toBe(true);
    if (!queryResult?.ok) return;
    expect(queryResult.data.size).toBe(0);
  });

  it('returns err with SUPABASE_ERROR when query fails', async () => {
    (supabase as any).from = vi.fn().mockReturnValue({
      select: vi.fn().mockResolvedValue({ data: null, error: { message: 'DB down' } }),
    });

    const { result } = renderHook(() => useRolePermissions(), {
      wrapper: createWrapper(),
    });

    await waitFor(() => expect(result.current.isSuccess).toBe(true));

    const queryResult = result.current.data;
    expect(queryResult?.ok).toBe(false);
    if (queryResult?.ok) return;
    expect(queryResult.error.code).toBe('SUPABASE_ERROR');
  });
});
