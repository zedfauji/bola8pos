**Step 1 — Create seed-reports.ts.**

File: `bar-pos/scripts/seed-reports.ts`

Follow the exact pattern from `seed-combos.ts` (eslint-disable at file level, supabase as any cast, VITE_SUPABASE_URL from .env.local).

The seed script inserts:
1. **7 days of combo sales** — Insert 7 `order_items` rows (one per day for the last 7 days) referencing a product where `is_combo = true`. Query for an existing combo product_id first.
2. **3 refunds** — Insert 3 rows into `refunds` referencing existing `payments.id`. Query for existing payment IDs first. Set `amount`, `reason = 'customer_request'`, `created_by` = admin profile ID.
3. **5 waitlist entries** — Insert into `waitlist_entries` with `status = 'seated'`, `seated_at` set to 30 minutes after `created_at`.

The script must handle `ON CONFLICT DO NOTHING` to be idempotent. Print "seed-reports: seeded combo sales, refunds, waitlist entries" on success.

```typescript
/* eslint-disable @typescript-eslint/no-explicit-any, @typescript-eslint/no-unsafe-assignment,
   @typescript-eslint/no-unsafe-call, @typescript-eslint/no-unsafe-member-access,
   @typescript-eslint/no-unsafe-argument */
import { createClient } from '@supabase/supabase-js';
import * as dotenv from 'dotenv';
import * as path from 'path';

dotenv.config({ path: path.resolve(process.cwd(), '.env.local') });

const supabaseUrl = process.env['VITE_SUPABASE_URL'] ?? '';
const serviceRoleKey = process.env['SUPABASE_SERVICE_ROLE_KEY'] ?? '';

if (!supabaseUrl || !serviceRoleKey) {
  console.error('seed-reports: missing VITE_SUPABASE_URL or SUPABASE_SERVICE_ROLE_KEY in .env.local');
  process.exit(1);
}

const db = createClient(supabaseUrl, serviceRoleKey) as any;

async function seedReports(): Promise<void> {
  // 1. Find a combo product
  const { data: combos } = await db.from('products').select('id').eq('is_combo', true).limit(1);
  const comboId = combos?.[0]?.id as string | undefined;

  // 2. Find recent orders to attach combo order_items to
  const { data: recentOrders } = await db
    .from('orders')
    .select('id')
    .eq('status', 'open')
    .order('created_at', { ascending: false })
    .limit(7);

  if (comboId && recentOrders && recentOrders.length > 0) {
    const comboItems = recentOrders.map((o: any, i: number) => ({
      order_id: o.id,
      product_id: comboId,
      quantity: 1,
      unit_price: 250,
      name: 'Cubeta Regular (seed)',
    }));
    await db.from('order_items').insert(comboItems).select();
    console.log('seed-reports: inserted combo order_items');
  }

  // 3. Find existing payments for refunds
  const { data: payments } = await db
    .from('payments')
    .select('id, tab_id')
    .order('created_at', { ascending: false })
    .limit(3);

  const { data: adminProfile } = await db
    .from('profiles')
    .select('id')
    .eq('role', 'admin')
    .limit(1);
  const adminId = adminProfile?.[0]?.id as string | undefined;

  if (payments && payments.length > 0 && adminId) {
    const refunds = payments.map((p: any) => ({
      original_payment_id: p.id,
      amount: 50,
      reason: 'customer_request',
      created_by: adminId,
    }));
    await db.from('refunds').insert(refunds).select();
    console.log('seed-reports: inserted refunds');
  }

  // 4. Insert waitlist entries
  const now = new Date();
  const waitlistEntries = Array.from({ length: 5 }, (_, i) => {
    const createdAt = new Date(now.getTime() - (i + 1) * 3600 * 1000);
    const seatedAt = new Date(createdAt.getTime() + 30 * 60 * 1000);
    return {
      party_name: `Test Party ${String(i + 1)}`,
      party_size: 2 + i,
      status: 'seated',
      created_at: createdAt.toISOString(),
      seated_at: seatedAt.toISOString(),
    };
  });
  await db.from('waitlist_entries').insert(waitlistEntries).select();
  console.log('seed-reports: inserted waitlist entries');

  console.log('seed-reports: done');
}

void seedReports();
```
