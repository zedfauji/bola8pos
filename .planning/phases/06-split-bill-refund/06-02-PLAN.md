**split_tab_by_item(p_parent_tab_id uuid, p_assignments jsonb)**
- p_assignments: `[{sub_tab_label: "Alice", order_item_ids: [uuid, ...]}, ...]`
- Validate: parent tab is OPEN (not split/paid/closed)
- Validate: all order_item_ids belong to parent tab (via order_items → orders → tab_id)
- Validate: no item_id appears in more than one assignment (ITEM_ASSIGNED_TWICE)
- For each assignment: INSERT tab row (parent_tab_id=parent, split_mode='item', split_label=label, status='open')
- For each assignment: INSERT orders row (tab_id=new_sub_tab_id, label=label, status='open')
- For each assignment: `UPDATE order_items SET order_id = new_order_id WHERE id = any(order_item_ids)`
- Cascade combo children: `UPDATE order_items SET order_id = new_order_id WHERE parent_order_item_id = ANY(order_item_ids)`
- UPDATE parent tab: status='split', split_mode='item'
- Return: `jsonb_agg(sub_tab_id)` as uuid[]
- Error codes: PARENT_TAB_PAID (status != 'open'), ITEM_NOT_IN_PARENT, ITEM_ASSIGNED_TWICE

**split_tab_evenly(p_parent_tab_id uuid, p_n integer)**
- Validate: parent tab is OPEN
- Validate: p_n >= 2
- Compute parent total: SELECT SUM(oi.unit_price * oi.qty) FROM order_items oi JOIN orders o ON o.id = oi.order_id WHERE o.tab_id = p_parent_tab_id
- per_payment_amount = FLOOR(total * 100 / p_n) / 100
- cents_remainder = total - per_payment_amount * (p_n - 1)
- Returns: `jsonb_build_object('per_payment_amount', per_payment_amount, 'cents_remainder', cents_remainder)`
- Does NOT create sub-tabs or modify parent status. Client inserts N payments.
- Error codes: PARENT_TAB_PAID

**split_tab_by_person(p_parent_tab_id uuid, p_n integer, p_assignments jsonb)**
- p_assignments: `[{sub_tab_label: "Person 1", order_item_ids: [uuid, ...]}, ...]`
- Unassigned items (not in any assignment) are identified and their total computed
- Each person sub-tab gets their assigned items + (unassigned_total / N, rounded)
- Validate: parent is OPEN, p_n >= 2, no ITEM_ASSIGNED_TWICE
- For each person (1..p_n): INSERT tab, INSERT orders row, UPDATE order_items SET order_id
- Cascade combo children same as by_item
- UPDATE parent: status='split', split_mode='by_person'
- Return: jsonb_agg of sub-tab IDs
- Unassigned items stay in the original parent orders (they appear in the parent for reference)

**split_tab_by_amount(p_parent_tab_id uuid, p_amounts jsonb)**
- p_amounts: `[{sub_tab_label: text, amount: numeric}, ...]`
- Validate: parent is OPEN, SUM(amounts) equals parent.total ± 0.01
- For each amount row: INSERT sub-tab (split_mode='by_amount', split_label=label), INSERT orders row
- Item assignment is proportional (allocate items greedily by amount until each sub-tab's threshold reached)
- UPDATE parent: status='split', split_mode='by_amount'
- Return: jsonb_agg of sub-tab IDs
- Include inline comment: "Note: by_amount mode loses item-level attribution — sub_tab stamped with split_mode='by_amount'"
- Error codes: PARENT_TAB_PAID, UNASSIGNED_ITEMS if proportional allocation leaves items unassigned

Full SQL structure:
```sql
-- =============================================================================
-- S4-03/04/05/06: Split tab RPCs
-- ... header comment ...
-- =============================================================================

BEGIN;

CREATE OR REPLACE FUNCTION split_tab_by_item(
  p_parent_tab_id uuid,
  p_assignments   jsonb
) RETURNS uuid[]
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_parent        record;
  v_assignment    jsonb;
  v_sub_tab_id    uuid;
  v_new_order_id  uuid;
  v_item_id       uuid;
  v_result_ids    uuid[] := '{}';
  v_assigned_ids  uuid[] := '{}';
BEGIN
  -- Lock and validate parent tab
  SELECT * INTO v_parent FROM tabs WHERE id = p_parent_tab_id FOR UPDATE;
  IF NOT FOUND OR v_parent.status != 'open' THEN
    RAISE EXCEPTION 'PARENT_TAB_PAID: tab % is not open (status: %)', p_parent_tab_id, v_parent.status;
  END IF;

  -- Validate no item assigned twice
  FOR v_assignment IN SELECT * FROM jsonb_array_elements(p_assignments) LOOP
    FOR v_item_id IN SELECT (elem)::uuid FROM jsonb_array_elements_text(v_assignment->'order_item_ids') AS elem LOOP
      IF v_item_id = ANY(v_assigned_ids) THEN
        RAISE EXCEPTION 'ITEM_ASSIGNED_TWICE: item % appears in multiple assignments', v_item_id;
      END IF;
      -- Verify item belongs to parent tab
      IF NOT EXISTS (
        SELECT 1 FROM order_items oi JOIN orders o ON o.id = oi.order_id
        WHERE oi.id = v_item_id AND o.tab_id = p_parent_tab_id
      ) THEN
        RAISE EXCEPTION 'ITEM_NOT_IN_PARENT: item % not found in tab %', v_item_id, p_parent_tab_id;
      END IF;
      v_assigned_ids := v_assigned_ids || v_item_id;
    END LOOP;
  END LOOP;

  -- Create sub-tabs and reassign items
  FOR v_assignment IN SELECT * FROM jsonb_array_elements(p_assignments) LOOP
    -- Create sub-tab row
    INSERT INTO tabs (parent_tab_id, split_mode, split_label, status, opened_by)
    SELECT p_parent_tab_id, 'item', v_assignment->>'sub_tab_label', 'open', v_parent.opened_by
    RETURNING id INTO v_sub_tab_id;

    -- Create order row for sub-tab
    INSERT INTO orders (tab_id, label, status)
    SELECT v_sub_tab_id, v_assignment->>'sub_tab_label', 'open'
    RETURNING id INTO v_new_order_id;

    -- Reassign order_items to new order
    FOR v_item_id IN SELECT (elem)::uuid FROM jsonb_array_elements_text(v_assignment->'order_item_ids') AS elem LOOP
      UPDATE order_items SET order_id = v_new_order_id WHERE id = v_item_id;
      -- Cascade combo children
      UPDATE order_items SET order_id = v_new_order_id WHERE parent_order_item_id = v_item_id;
    END LOOP;

    v_result_ids := v_result_ids || v_sub_tab_id;
  END LOOP;

  -- Mark parent as split
  UPDATE tabs SET status = 'split', split_mode = 'item', updated_at = now()
  WHERE id = p_parent_tab_id;

  RETURN v_result_ids;
END;
$$;

CREATE OR REPLACE FUNCTION split_tab_evenly(
  p_parent_tab_id uuid,
  p_n             integer
) RETURNS jsonb
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_parent      record;
  v_total       numeric;
  v_base        numeric;
  v_remainder   numeric;
BEGIN
  SELECT * INTO v_parent FROM tabs WHERE id = p_parent_tab_id;
  IF NOT FOUND OR v_parent.status != 'open' THEN
    RAISE EXCEPTION 'PARENT_TAB_PAID: tab % is not open', p_parent_tab_id;
  END IF;
  IF p_n < 2 THEN
    RAISE EXCEPTION 'VALIDATION_ERROR: n must be at least 2, got %', p_n;
  END IF;

  SELECT COALESCE(SUM(oi.unit_price * oi.qty), 0)
  INTO v_total
  FROM order_items oi
  JOIN orders o ON o.id = oi.order_id
  WHERE o.tab_id = p_parent_tab_id;

  -- Floor to cent, last payment absorbs remainder
  v_base      := FLOOR(v_total * 100 / p_n) / 100;
  v_remainder := v_total - v_base * (p_n - 1);

  RETURN jsonb_build_object(
    'per_payment_amount', v_base,
    'cents_remainder',    v_remainder - v_base
  );
END;
$$;

CREATE OR REPLACE FUNCTION split_tab_by_person(
  p_parent_tab_id uuid,
  p_n             integer,
  p_assignments   jsonb
) RETURNS uuid[]
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_parent        record;
  v_assignment    jsonb;
  v_sub_tab_id    uuid;
  v_new_order_id  uuid;
  v_item_id       uuid;
  v_result_ids    uuid[] := '{}';
  v_assigned_ids  uuid[] := '{}';
  v_i             integer := 1;
BEGIN
  SELECT * INTO v_parent FROM tabs WHERE id = p_parent_tab_id FOR UPDATE;
  IF NOT FOUND OR v_parent.status != 'open' THEN
    RAISE EXCEPTION 'PARENT_TAB_PAID: tab % is not open', p_parent_tab_id;
  END IF;
  IF p_n < 2 THEN
    RAISE EXCEPTION 'VALIDATION_ERROR: n must be at least 2, got %', p_n;
  END IF;

  -- Validate no duplicates
  FOR v_assignment IN SELECT * FROM jsonb_array_elements(p_assignments) LOOP
    FOR v_item_id IN SELECT (elem)::uuid FROM jsonb_array_elements_text(v_assignment->'order_item_ids') AS elem LOOP
      IF v_item_id = ANY(v_assigned_ids) THEN
        RAISE EXCEPTION 'ITEM_ASSIGNED_TWICE: item % assigned to multiple persons', v_item_id;
      END IF;
      v_assigned_ids := v_assigned_ids || v_item_id;
    END LOOP;
  END LOOP;

  -- Pad assignments to p_n if fewer provided
  WHILE jsonb_array_length(p_assignments) < p_n LOOP
    p_assignments := p_assignments || jsonb_build_object('sub_tab_label', 'Person ' || (jsonb_array_length(p_assignments) + 1), 'order_item_ids', '[]'::jsonb);
  END LOOP;

  FOR v_assignment IN SELECT * FROM jsonb_array_elements(p_assignments) LOOP
    INSERT INTO tabs (parent_tab_id, split_mode, split_label, status, opened_by)
    SELECT p_parent_tab_id, 'by_person', v_assignment->>'sub_tab_label', 'open', v_parent.opened_by
    RETURNING id INTO v_sub_tab_id;

    INSERT INTO orders (tab_id, label, status)
    SELECT v_sub_tab_id, v_assignment->>'sub_tab_label', 'open'
    RETURNING id INTO v_new_order_id;

    FOR v_item_id IN SELECT (elem)::uuid FROM jsonb_array_elements_text(v_assignment->'order_item_ids') AS elem LOOP
      UPDATE order_items SET order_id = v_new_order_id WHERE id = v_item_id;
      UPDATE order_items SET order_id = v_new_order_id WHERE parent_order_item_id = v_item_id;
    END LOOP;

    v_result_ids := v_result_ids || v_sub_tab_id;
    v_i := v_i + 1;
  END LOOP;

  UPDATE tabs SET status = 'split', split_mode = 'by_person', updated_at = now()
  WHERE id = p_parent_tab_id;

  RETURN v_result_ids;
END;
$$;

CREATE OR REPLACE FUNCTION split_tab_by_amount(
  p_parent_tab_id uuid,
  p_amounts       jsonb  -- [{sub_tab_label: text, amount: numeric}, ...]
) RETURNS uuid[]
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_parent       record;
  v_total        numeric;
  v_amounts_sum  numeric;
  v_amount_row   jsonb;
  v_sub_tab_id   uuid;
  v_new_order_id uuid;
  v_result_ids   uuid[] := '{}';
BEGIN
  SELECT * INTO v_parent FROM tabs WHERE id = p_parent_tab_id FOR UPDATE;
  IF NOT FOUND OR v_parent.status != 'open' THEN
    RAISE EXCEPTION 'PARENT_TAB_PAID: tab % is not open', p_parent_tab_id;
  END IF;

  SELECT COALESCE(SUM(oi.unit_price * oi.qty), 0) INTO v_total
  FROM order_items oi JOIN orders o ON o.id = oi.order_id WHERE o.tab_id = p_parent_tab_id;

  SELECT SUM((row->>'amount')::numeric) INTO v_amounts_sum
  FROM jsonb_array_elements(p_amounts) AS row;

  IF ABS(v_amounts_sum - v_total) > 0.01 THEN
    RAISE EXCEPTION 'VALIDATION_ERROR: amounts sum % does not match tab total % (±0.01 allowed)', v_amounts_sum, v_total;
  END IF;

  -- Note: by_amount mode does not reassign order_items (item-level reporting loses precision;
  --       sub_tab stamped with split_mode=''by_amount'' for report awareness)
  FOR v_amount_row IN SELECT * FROM jsonb_array_elements(p_amounts) LOOP
    INSERT INTO tabs (parent_tab_id, split_mode, split_label, status, opened_by)
    SELECT p_parent_tab_id, 'by_amount', v_amount_row->>'sub_tab_label', 'open', v_parent.opened_by
    RETURNING id INTO v_sub_tab_id;

    INSERT INTO orders (tab_id, label, status)
    SELECT v_sub_tab_id, v_amount_row->>'sub_tab_label', 'open'
    RETURNING id INTO v_new_order_id;

    v_result_ids := v_result_ids || v_sub_tab_id;
  END LOOP;

  UPDATE tabs SET status = 'split', split_mode = 'by_amount', updated_at = now()
  WHERE id = p_parent_tab_id;

  RETURN v_result_ids;
END;
$$;

COMMIT;
```
