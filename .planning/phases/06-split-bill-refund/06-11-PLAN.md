import { createClient } from '@supabase/supabase-js';
import { expect, test } from '@playwright/test';
import { loginAs, logout } from './helpers/auth';
import { requireIntegrationEnv } from './helpers/requireEnv';
import { openCaja, resetTestState } from './helpers/supabase';

function getServiceClient() {
  const url = process.env['VITE_SUPABASE_URL']!;
  const key = process.env['SUPABASE_SERVICE_ROLE_KEY']!;
  return createClient(url, key, {
    auth: { persistSession: false, autoRefreshToken: false },
  }) as any;
}

/** Seed a paid tab with itemCount items at unitPrice each. Returns { tabId, paymentId, orderItemIds }. */
async function seedPaidTab(db: any, itemCount: number, unitPrice: number) {
  const { data: profile } = await db
    .from('profiles').select('id').eq('role', 'admin').limit(1).single();

  const { data: tab } = await db.from('tabs').insert({
    label: 'E2E Refund Test',
    status: 'open',
    opened_by: profile.id,
  }).select('id').single();

  const { data: order } = await db.from('orders').insert({
    tab_id: tab.id,
    status: 'open',
  }).select('id').single();

  const { data: product } = await db.from('products').select('id').limit(1).single();

  const orderItemInserts = Array.from({ length: itemCount }, () => ({
    order_id: order.id,
    product_id: product.id,
    qty: 1,
    unit_price: unitPrice,
  }));
  const { data: orderItems } = await db.from('order_items').insert(orderItemInserts).select('id');

  await db.from('tabs')
    .update({ status: 'paid', closed_at: new Date().toISOString() })
    .eq('id', tab.id);

  const { data: payment } = await db.from('payments').insert({
    tab_id: tab.id,
    amount: itemCount * unitPrice,
    tip_amount: 0,
    method: 'cash',
    is_refund: false,
  }).select('id').single();

  return {
    tabId: tab.id as string,
    paymentId: payment.id as string,
    orderItemIds: (orderItems as { id: string }[]).map(i => i.id),
  };
}

test('T5: REFUND_EXCEEDS_ORIGINAL blocks double-refund of fully-refunded payment', async ({ page }) => {
  test.info().annotations.push({ type: 'requirement', description: 'S4-07, S4-19' });
  const db = getServiceClient();

  // Seed paid tab with 2 items
  const { tabId, paymentId, orderItemIds } = await seedPaidTab(db, 2, 10.00);

  // Pre-seed a full refund via DB
  const { data: profile } = await db
    .from('profiles').select('id').eq('role', 'admin').limit(1).single();
  const { data: existingRefund } = await db.from('refunds').insert({
    original_payment_id: paymentId,
    reason: 'wrong_order',
    amount: 20.00,
    created_by: profile.id,
  }).select('id').single();
  await db.from('refund_items').insert(
    orderItemIds.map((id: string) => ({
      refund_id: existingRefund.id,
      order_item_id: id,
      qty: 1,
      amount: 10.00,
      restock: false,
    }))
  );
  await db.from('payments').insert({
    tab_id: tabId,
    amount: -20.00,
    tip_amount: 0,
    method: 'cash',
    is_refund: true,
    refund_id: existingRefund.id,
  });

  await loginAs(page, 'admin');
  await page.goto('/payments');

  const refundBtn = page.getByRole('button', { name: 'Refund' }).first();
  await expect(refundBtn).toBeVisible();
  await refundBtn.click();
  await expect(page.getByRole('dialog')).toBeVisible();

  // All items fully refunded: checkboxes should be disabled, Request approval disabled
  const requestBtn = page.getByRole('button', { name: /request approval/i });
  const firstCb = page.getByRole('checkbox').first();
  const cbDisabled = await firstCb.isDisabled();

  if (cbDisabled) {
    // UI correctly prevents submission when all items fully refunded
    await expect(requestBtn).toBeDisabled();
  } else {
    // Fallback: UI allows selection -- submit and expect RPC to block
    await firstCb.check();
    await page.getByRole('combobox').click();
    await page.getByRole('option', { name: /wrong.?order/i }).click();
    await requestBtn.click();
    const pinInput = page
      .locator('input[type="password"]')
      .or(page.getByRole('textbox', { name: /pin/i }))
      .first();
    await pinInput.fill('0000');
    await page.getByRole('button', { name: /confirm|submit|approve/i }).last().click();
    await expect(
      page.getByText(/exceeds.*refundable/i).or(page.getByText(/exceeds the remaining/i))
    ).toBeVisible({ timeout: 8000 });
  }
});

test('T6: refund remaining items with restock=false -- no ledger change', async ({ page }) => {
  test.info().annotations.push({ type: 'requirement', description: 'S4-07, S4-19' });
  const db = getServiceClient();

  // Seed: paid tab with 5 items; pre-refund items[0] and [1] via DB
  const { tabId, paymentId, orderItemIds } = await seedPaidTab(db, 5, 10.00);
  const { data: profile } = await db
    .from('profiles').select('id').eq('role', 'admin').limit(1).single();
  const { data: partialRefund } = await db.from('refunds').insert({
    original_payment_id: paymentId,
    reason: 'wrong_order',
    amount: 20.00,
    created_by: profile.id,
  }).select('id').single();
  await db.from('refund_items').insert([
    { refund_id: partialRefund.id, order_item_id: orderItemIds[0], qty: 1, amount: 10.00, restock: false },
    { refund_id: partialRefund.id, order_item_id: orderItemIds[1], qty: 1, amount: 10.00, restock: false },
  ]);
  await db.from('payments').insert({
    tab_id: tabId,
    amount: -20.00,
    tip_amount: 0,
    method: 'cash',
    is_refund: true,
    refund_id: partialRefund.id,
  });

  await loginAs(page, 'admin');
  await page.goto('/payments');

  const refundBtn = page.getByRole('button', { name: 'Refund' }).first();
  await expect(refundBtn).toBeVisible();
  await refundBtn.click();
  await expect(page.getByRole('dialog')).toBeVisible();

  // Select remaining 3 items (items[0] and [1] are fully refunded -- disabled)
  const checkboxes = page.getByRole('checkbox');
  const total = await checkboxes.count();
  for (let i = 0; i < total; i++) {
    const cb = checkboxes.nth(i);
    if (!(await cb.isDisabled())) {
      await cb.check();
    }
  }

  // Set restock=false: uncheck any restock toggles
  const restockCheckboxes = page.getByLabel(/restock/i);
  const restockCount = await restockCheckboxes.count();
  for (let i = 0; i < restockCount; i++) {
    const rc = restockCheckboxes.nth(i);
    if (await rc.isChecked()) {
      await rc.uncheck();
    }
  }

  await page.getByRole('combobox').click();
  await page.getByRole('option', { name: /wrong.?order/i }).click();

  await page.getByRole('button', { name: /request approval/i }).click();
  const pinInput = page
    .locator('input[type="password"]')
    .or(page.getByRole('textbox', { name: /pin/i }))
    .first();
  await pinInput.fill('0000');
  await page.getByRole('button', { name: /confirm|submit|approve/i }).last().click();

  await expect(page.getByText(/refund.*processed/i)).toBeVisible({ timeout: 8000 });

  // DB: second refund row for remaining 3 items ($30)
  const { data: secondRefund } = await db
    .from('refunds')
    .select('id, amount')
    .eq('original_payment_id', paymentId)
    .neq('id', partialRefund.id)
    .single();
  expect(secondRefund).not.toBeNull();
  expect(Number(secondRefund.amount)).toBeCloseTo(30.00, 2);

  // restock=false: no ledger change (Phase 4 deplete_for_order_item stub returns gracefully)
  const { data: refundItems } = await db
    .from('refund_items')
    .select('restock')
    .eq('refund_id', secondRefund.id);
  expect((refundItems as { restock: boolean }[]).every(i => i.restock === false)).toBe(true);
});
