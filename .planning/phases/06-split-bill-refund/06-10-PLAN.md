/**
 * Integration tests for split-tab RPCs (S4-16)
 * Requires: VITE_SUPABASE_URL + SUPABASE_SERVICE_ROLE_KEY env vars
 * Run: cd bar-pos && npx vitest run src/features/split-tab/split-tab.integration.test.ts
 */
import { createClient } from '@supabase/supabase-js';
import { afterEach, beforeEach, describe, expect, it } from 'vitest';

const hasEnv =
  typeof process.env['VITE_SUPABASE_URL'] === 'string' &&
  typeof process.env['SUPABASE_SERVICE_ROLE_KEY'] === 'string';

const itInt = hasEnv ? it : it.skip;

function getDb() {
  const url = process.env['VITE_SUPABASE_URL']!;
  const key = process.env['SUPABASE_SERVICE_ROLE_KEY']!;
  return createClient(url, key, {
    auth: { persistSession: false, autoRefreshToken: false },
  }) as any;
}

// ── Seed helpers ────────────────────────────────────────────────────────────

async function seedTabWithItems(db: any, itemCount: number) {
  // Get a valid staff profile for opened_by / processed_by
  const { data: profile } = await db
    .from('profiles')
    .select('id, role')
    .limit(1)
    .single();
  const staffId = profile.id;

  // Get active caja session (required by tabs FK if exists)
  const { data: caja } = await db
    .from('caja_sessions')
    .select('id')
    .eq('status', 'open')
    .limit(1)
    .maybeSingle();

  // Create tab
  const { data: tab } = await db
    .from('tabs')
    .insert({ label: `Test Split Tab ${Date.now()}`, status: 'open', opened_by: staffId })
    .select()
    .single();

  // Create order
  const { data: order } = await db
    .from('orders')
    .insert({ tab_id: tab.id, label: 'Test Order', status: 'open' })
    .select()
    .single();

  // Get a test product
  const { data: product } = await db
    .from('products')
    .select('id, price')
    .eq('is_active', true)
    .limit(1)
    .single();

  // Create order_items
  const itemInserts = Array.from({ length: itemCount }, (_, i) => ({
    order_id: order.id,
    product_id: product.id,
    qty: 1,
    unit_price: product.price ?? 10.00,
    product_name: `Test Item ${i + 1}`,
  }));

  const { data: items } = await db
    .from('order_items')
    .insert(itemInserts)
    .select();

  return { tab, order, items: items as any[], staffId };
}

async function cleanup(db: any, tabId: string) {
  // Cascade: sub-tabs + their orders + order_items cleaned up via FK cascades
  // Order: payments → order_items → orders → tabs (child first)
  await db.from('payments').delete().eq('tab_id', tabId);
  // Delete sub-tabs first
  const { data: subTabs } = await db.from('tabs').select('id').eq('parent_tab_id', tabId);
  for (const sub of subTabs ?? []) {
    await db.from('payments').delete().eq('tab_id', sub.id);
    const { data: subOrders } = await db.from('orders').select('id').eq('tab_id', sub.id);
    for (const o of subOrders ?? []) {
      await db.from('order_items').delete().eq('order_id', o.id);
    }
    await db.from('orders').delete().eq('tab_id', sub.id);
  }
  await db.from('tabs').delete().eq('parent_tab_id', tabId);
  const { data: orders } = await db.from('orders').select('id').eq('tab_id', tabId);
  for (const o of orders ?? []) {
    await db.from('order_items').delete().eq('order_id', o.id);
  }
  await db.from('orders').delete().eq('tab_id', tabId);
  await db.from('tabs').delete().eq('id', tabId);
}

// ── Tests ────────────────────────────────────────────────────────────────────

describe('split_tab_by_item integration (S4-16)', () => {
  let db: any;
  let testTabId: string;

  beforeEach(() => { db = getDb(); });
  afterEach(async () => {
    if (testTabId) await cleanup(db, testTabId).catch(() => undefined);
  });

  itInt('creates 3 sub-tabs and reassigns 10 items across 3 assignments', async () => {
    const { tab, items } = await seedTabWithItems(db, 10);
    testTabId = tab.id;

    // Distribute: Alice=items[0..3], Bob=items[4..6], Charlie=items[7..9]
    const assignments = [
      { sub_tab_label: 'Alice',   order_item_ids: items.slice(0, 4).map((i: any) => i.id) },
      { sub_tab_label: 'Bob',     order_item_ids: items.slice(4, 7).map((i: any) => i.id) },
      { sub_tab_label: 'Charlie', order_item_ids: items.slice(7).map((i: any) => i.id) },
    ];

    const { data: subTabIds, error } = await db.rpc('split_tab_by_item', {
      p_parent_tab_id: tab.id,
      p_assignments: assignments,
    });

    expect(error).toBeNull();
    expect(subTabIds).toHaveLength(3);

    // Parent is now 'split'
    const { data: parent } = await db.from('tabs').select('status, split_mode').eq('id', tab.id).single();
    expect(parent.status).toBe('split');
    expect(parent.split_mode).toBe('item');
  });

  itInt('parent auto-closes when all sub-tabs paid', async () => {
    const { tab, items, staffId } = await seedTabWithItems(db, 2);
    testTabId = tab.id;

    const { data: subTabIds } = await db.rpc('split_tab_by_item', {
      p_parent_tab_id: tab.id,
      p_assignments: [
        { sub_tab_label: 'A', order_item_ids: [items[0].id] },
        { sub_tab_label: 'B', order_item_ids: [items[1].id] },
      ],
    });

    // Pay first sub-tab
    await db.from('payments').insert({
      tab_id: subTabIds[0], amount: 10, tip_amount: 0, method: 'cash',
      processed_by: staffId, processed_at: new Date().toISOString(),
    });

    let { data: parent } = await db.from('tabs').select('status').eq('id', tab.id).single();
    expect(parent.status).toBe('split'); // still split

    // Pay second sub-tab → trigger fires
    await db.from('payments').insert({
      tab_id: subTabIds[1], amount: 10, tip_amount: 0, method: 'cash',
      processed_by: staffId, processed_at: new Date().toISOString(),
    });

    ({ data: parent } = await db.from('tabs').select('status').eq('id', tab.id).single());
    expect(parent.status).toBe('paid'); // auto-closed!
  });

  itInt('blocks ITEM_ASSIGNED_TWICE', async () => {
    const { tab, items } = await seedTabWithItems(db, 3);
    testTabId = tab.id;

    const { error } = await db.rpc('split_tab_by_item', {
      p_parent_tab_id: tab.id,
      p_assignments: [
        { sub_tab_label: 'A', order_item_ids: [items[0].id, items[1].id] },
        { sub_tab_label: 'B', order_item_ids: [items[1].id, items[2].id] }, // items[1] duplicated
      ],
    });

    expect(error).not.toBeNull();
    expect(error.message).toContain('ITEM_ASSIGNED_TWICE');
  });
});

/**
 * Integration tests for process_refund RPC (S4-17)
 * Requires: VITE_SUPABASE_URL + SUPABASE_SERVICE_ROLE_KEY env vars
 * Run: cd bar-pos && npx vitest run src/features/process-refund/process-refund.integration.test.ts
 */
import { createClient } from '@supabase/supabase-js';
import { afterEach, beforeEach, describe, expect, it } from 'vitest';

const hasEnv =
  typeof process.env['VITE_SUPABASE_URL'] === 'string' &&
  typeof process.env['SUPABASE_SERVICE_ROLE_KEY'] === 'string';

const itInt = hasEnv ? it : it.skip;

function getDb() {
  const url = process.env['VITE_SUPABASE_URL']!;
  const key = process.env['SUPABASE_SERVICE_ROLE_KEY']!;
  return createClient(url, key, {
    auth: { persistSession: false, autoRefreshToken: false },
  }) as any;
}

async function seedPaidTab(db: any) {
  const { data: profile } = await db
    .from('profiles')
    .select('id, role')
    .in('role', ['manager', 'admin'])
    .limit(1)
    .single();
  const staffId = profile.id;

  const { data: tab } = await db
    .from('tabs')
    .insert({ label: `Refund Test Tab ${Date.now()}`, status: 'open', opened_by: staffId })
    .select()
    .single();

  const { data: order } = await db
    .from('orders')
    .insert({ tab_id: tab.id, label: 'Test', status: 'open' })
    .select()
    .single();

  const { data: product } = await db
    .from('products')
    .select('id, price')
    .eq('is_active', true)
    .limit(1)
    .single();

  const { data: items } = await db
    .from('order_items')
    .insert(
      Array.from({ length: 5 }, (_, i) => ({
        order_id: order.id,
        product_id: product.id,
        qty: 1,
        unit_price: 10.00,
        product_name: `Item ${i + 1}`,
      }))
    )
    .select();

  // Insert payment (marks tab as paid)
  const { data: payment } = await db
    .from('payments')
    .insert({
      tab_id: tab.id,
      amount: 50.00,
      tip_amount: 0,
      method: 'cash',
      processed_by: staffId,
      processed_at: new Date().toISOString(),
    })
    .select()
    .single();

  await db.from('tabs').update({ status: 'paid', closed_at: new Date().toISOString() }).eq('id', tab.id);

  return { tab, order, items: items as any[], payment, staffId };
}

async function cleanupRefund(db: any, tabId: string) {
  const { data: payments } = await db.from('payments').select('id').eq('tab_id', tabId);
  for (const p of payments ?? []) {
    const { data: refunds } = await db.from('refunds').select('id').eq('original_payment_id', p.id);
    for (const r of refunds ?? []) {
      await db.from('refund_items').delete().eq('refund_id', r.id);
    }
    await db.from('refunds').delete().eq('original_payment_id', p.id);
  }
  await db.from('payments').delete().eq('tab_id', tabId);
  const { data: orders } = await db.from('orders').select('id').eq('tab_id', tabId);
  for (const o of orders ?? []) {
    await db.from('order_items').delete().eq('order_id', o.id);
  }
  await db.from('orders').delete().eq('tab_id', tabId);
  await db.from('tabs').delete().eq('id', tabId);
}

describe('process_refund integration (S4-17)', () => {
  let db: any;
  let testTabId: string;

  beforeEach(() => { db = getDb(); });
  afterEach(async () => {
    if (testTabId) await cleanupRefund(db, testTabId).catch(() => undefined);
  });

  itInt('inserts refund + refund_items + negative payment row for 2-item refund', async () => {
    const { tab, items, payment } = await seedPaidTab(db);
    testTabId = tab.id;

    const refundItems = [
      { order_item_id: items[0].id, qty: 1, amount: 10.00, restock: true },
      { order_item_id: items[1].id, qty: 1, amount: 10.00, restock: true },
    ];

    const { data: refundId, error } = await db.rpc('process_refund', {
      p_original_payment_id: payment.id,
      p_items: refundItems,
      p_reason: 'wrong_order',
      p_manager_pin: '',
    });

    expect(error).toBeNull();
    expect(refundId).toBeTruthy();

    // Verify refund row
    const { data: refund } = await db.from('refunds').select('*').eq('id', refundId).single();
    expect(refund.amount).toBe(20.00);
    expect(refund.reason).toBe('wrong_order');

    // Verify refund_items rows
    const { data: rItems } = await db.from('refund_items').select('*').eq('refund_id', refundId);
    expect(rItems).toHaveLength(2);

    // Verify negative payment row
    const { data: negPayment } = await db
      .from('payments')
      .select('*')
      .eq('refund_id', refundId)
      .eq('is_refund', true)
      .single();
    expect(negPayment.amount).toBe(-20.00);
    expect(negPayment.is_refund).toBe(true);
  });

  itInt('blocks REFUND_EXCEEDS_ORIGINAL on double-refund of same items', async () => {
    const { tab, items, payment } = await seedPaidTab(db);
    testTabId = tab.id;

    const refundItems = [
      { order_item_id: items[0].id, qty: 1, amount: 10.00, restock: false },
    ];

    // First refund — succeeds
    await db.rpc('process_refund', {
      p_original_payment_id: payment.id,
      p_items: refundItems,
      p_reason: 'billing_error',
      p_manager_pin: '',
    });

    // Second refund of same amount — should fail (already refunded + this exceeds original)
    const { error } = await db.rpc('process_refund', {
      p_original_payment_id: payment.id,
      p_items: [
        ...refundItems,
        ...items.slice(1, 5).map((i: any) => ({ order_item_id: i.id, qty: 1, amount: 10.00, restock: false })),
      ], // total = 50.00, but original = 50.00 and 10.00 already refunded → exceeds by 10.00
      p_reason: 'billing_error',
      p_manager_pin: '',
    });

    expect(error).not.toBeNull();
    expect(error.message).toContain('REFUND_EXCEEDS_ORIGINAL');
  });
});
