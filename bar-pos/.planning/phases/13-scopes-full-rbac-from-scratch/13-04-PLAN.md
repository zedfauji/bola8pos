/* eslint-disable @typescript-eslint/no-explicit-any, @typescript-eslint/no-unsafe-assignment,

// Pre-regen cast: role_permissions table added in Phase 13; supabase.types.ts extended manually.
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { supabase } from '@shared/lib/supabase';
import { logger } from '@shared/lib/logger-instance';
import { err, ok, type Result } from '@shared/lib/result';
import type { StaffAction, StaffRole } from '@shared/lib/rbac';
import { rbacKeys } from '@entities/rbac';

const db = supabase as any;

export function useMutationTogglePermission() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: async ({
      role,
      action,
      enabled,
    }: {
      role: StaffRole;
      action: StaffAction;
      enabled: boolean;
    }): Promise<Result<null>> => {
      if (enabled) {
        const { error } = await db
          .from('role_permissions')
          .insert({ role, action })
          .select()
          .single();
        if (error) {
          logger.error('toggle-permission.enable.failed', { role, action, error });
          return err({
            code: 'SUPABASE_ERROR' as const,
            message: (error as { message?: string }).message ?? 'Failed to enable permission',
          });
        }
        return ok(null);
      } else {
        const { error } = await db
          .from('role_permissions')
          .delete()
          .eq('role', role)
          .eq('action', action);
        if (error) {
          logger.error('toggle-permission.disable.failed', { role, action, error });
          return err({
            code: 'SUPABASE_ERROR' as const,
            message: (error as { message?: string }).message ?? 'Failed to disable permission',
          });
        }
        return ok(null);
      }
    },
    onSuccess: result => {
      if (result.ok) void queryClient.invalidateQueries({ queryKey: rbacKeys.list() });
    },
  });
}
