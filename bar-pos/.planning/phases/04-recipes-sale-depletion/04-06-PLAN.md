```typescript
import { describe, it, expect } from 'vitest';
import * as fc from 'fast-check';
import { computeDepletion } from '@shared/lib/domain-helpers';
import type { RecipeWithItems } from '@shared/lib/domain';

function makeRecipe(items: { ingredientId: string; qty: number }[], yieldQty = 1): RecipeWithItems {
  return {
    id: 'r1', productId: 'p1', yieldQty, notes: null,
    items: items.map((item, i) => ({ id: `ri${i}`, recipeId: 'r1', ...item })),
  };
}

describe('computeDepletion', () => {
  it('returns negative deltas for sale (direction=+1)', () => {
    const recipe = makeRecipe([{ ingredientId: 'beer', qty: 0.355 }, { ingredientId: 'lime', qty: 0.5 }]);
    const deltas = computeDepletion(recipe, 1, 1);
    expect(deltas.get('beer')).toBeCloseTo(-0.355);
    expect(deltas.get('lime')).toBeCloseTo(-0.5);
  });

  it('returns exact additive inverse for void (direction=-1)', () => {
    const recipe = makeRecipe([{ ingredientId: 'beer', qty: 0.355 }]);
    const sale = computeDepletion(recipe, 1, 1);
    const void_ = computeDepletion(recipe, 1, -1);
    expect((void_.get('beer') ?? 0) + (sale.get('beer') ?? 0)).toBeCloseTo(0);
  });

  it('returns empty Map for recipe with no items', () => {
    const recipe = makeRecipe([]);
    expect(computeDepletion(recipe, 1, 1).size).toBe(0);
  });

  it('scales linearly with orderQty', () => {
    const recipe = makeRecipe([{ ingredientId: 'wing', qty: 6 }]);
    const single = computeDepletion(recipe, 1, 1).get('wing') ?? 0;
    const triple = computeDepletion(recipe, 3, 1).get('wing') ?? 0;
    expect(triple).toBeCloseTo(single * 3);
  });

  it('P6: sum of |deltas| equals orderQty × (Σitem.qty / yieldQty)', () => {
    const itemArb = fc.record({
      ingredientId: fc.uuid(),
      qty: fc.float({ min: 0.001, max: 100, noNaN: true }),
    });
    const recipeArb = fc.record({
      items: fc.array(itemArb, { minLength: 1, maxLength: 10 }),
      yieldQty: fc.float({ min: 0.001, max: 100, noNaN: true }),
    });
    const orderQtyArb = fc.float({ min: 0.001, max: 50, noNaN: true });

    fc.assert(
      fc.property(recipeArb, orderQtyArb, ({ items, yieldQty }, orderQty) => {
        const recipe = makeRecipe(items, yieldQty);
        const deltas = computeDepletion(recipe, orderQty, 1);
        const expectedSum = items.reduce((acc, item) => acc + (orderQty * item.qty) / yieldQty, 0);
        const actualSum = Array.from(deltas.values()).reduce((acc, d) => acc + Math.abs(d), 0);
        return Math.abs(actualSum - expectedSum) < 1e-6;
      }),
      { numRuns: 500 }
    );
  });
});

import { describe, it, expect, beforeAll, afterAll } from 'vitest';
import { createClient } from '@supabase/supabase-js';

const url = import.meta.env.VITE_SUPABASE_URL as string | undefined;
const serviceKey = process.env['SUPABASE_SERVICE_ROLE_KEY'];
const skip = !url || !serviceKey;

describe.skipIf(skip)('Depletion integration', () => {
  // eslint-disable-next-line @typescript-eslint/no-non-null-assertion
  const db = createClient(url!, serviceKey!);

  let productId: string;
  let ingredientAId: string;
  let ingredientBId: string;
  let recipeId: string;
  let tabId: string;
  let orderItemId: string;

  beforeAll(async () => {
    // Create product
    const { data: prod } = await db.from('products').insert({ name: '__test_alitas__', price: 5.00, category_id: null }).select('id').single();
    productId = (prod as { id: string }).id;

    // Create 2 ingredients with enough stock
    const { data: ingA } = await db.from('ingredients').insert({ name: '__test_wing__', unit: 'pc', quantity_on_hand: 100, reorder_point: 10 }).select('id').single();
    const { data: ingB } = await db.from('ingredients').insert({ name: '__test_sauce__', unit: 'oz', quantity_on_hand: 200, reorder_point: 5 }).select('id').single();
    ingredientAId = (ingA as { id: string }).id;
    ingredientBId = (ingB as { id: string }).id;

    // Create recipe: 6 wings + 2 oz sauce per 1 serving
    const { data: rec } = await db.from('recipes').insert({ product_id: productId, yield_qty: 1 }).select('id').single();
    recipeId = (rec as { id: string }).id;
    await db.from('recipe_items').insert([
      { recipe_id: recipeId, ingredient_id: ingredientAId, qty: 6 },
      { recipe_id: recipeId, ingredient_id: ingredientBId, qty: 2 },
    ]);

    // Create a tab and open caja (required by caja guard in create_order_with_items)
    // Use admin credentials stored in env, or use the existing open caja from test state
    // NOTE: Requires an open caja session. If none exists in dev DB, this beforeAll will need to open one.
    // For simplicity: use the db to directly insert an order_item bypassing create_order_with_items,
    // then call deplete_for_order_item directly.
    const { data: tab } = await db.from('tabs').insert({ bartender_id: null, customer_name: '__test_tab__', status: 'open' }).select('id').single();
    tabId = (tab as { id: string }).id;

    // Insert order for test product
    const { data: order } = await db.from('orders').insert({ tab_id: tabId, status: 'pending' }).select('id').single();
    const { data: item } = await db
      .from('order_items')
      .insert({ order_id: (order as { id: string }).id, product_id: productId, quantity: 1, unit_price: 5.00 })
      .select('id')
      .single();
    orderItemId = (item as { id: string }).id;
  });

  afterAll(async () => {
    // Cleanup: delete in reverse FK order
    await db.from('stock_movements').delete().in('ingredient_id', [ingredientAId, ingredientBId]);
    await db.from('audit_log').delete().eq('entity_type', 'order_item').eq('entity_id', orderItemId);
    await db.from('order_items').delete().eq('id', orderItemId);
    await db.from('orders').delete().eq('tab_id', tabId);
    await db.from('tabs').delete().eq('id', tabId);
    await db.from('recipe_items').delete().eq('recipe_id', recipeId);
    await db.from('recipes').delete().eq('id', recipeId);
    await db.from('ingredients').delete().in('id', [ingredientAId, ingredientBId]);
    await db.from('products').delete().eq('id', productId);
  });

  it('I1: deplete_for_order_item writes 2 negative stock_movement rows', async () => {
    await db.rpc('deplete_for_order_item', { p_order_item_id: orderItemId, p_direction: 1 });
    const { data } = await db.from('stock_movements').select('*').in('ingredient_id', [ingredientAId, ingredientBId]).order('created_at');
    expect(data).toHaveLength(2);
    const wingRow = (data as { ingredient_id: string; delta: number }[]).find(r => r.ingredient_id === ingredientAId);
    expect(wingRow?.delta).toBeCloseTo(-6);
  });

  it('I2: deplete_for_order_item direction=-1 writes 2 positive reversal rows', async () => {
    await db.rpc('deplete_for_order_item', { p_order_item_id: orderItemId, p_direction: -1 });
    const { data } = await db.from('stock_movements').select('*').in('ingredient_id', [ingredientAId, ingredientBId]).eq('delta', 6).order('created_at');
    expect(data).toHaveLength(2);
  });

  it('I3: deplete when stock=0 raises INVENTORY_NEGATIVE', async () => {
    await db.from('ingredients').update({ quantity_on_hand: 0 }).in('id', [ingredientAId, ingredientBId]);
    const { error } = await db.rpc('deplete_for_order_item', { p_order_item_id: orderItemId, p_direction: 1 });
    expect(error).not.toBeNull();
    expect(error?.message ?? '').toMatch(/INVENTORY_NEGATIVE/i);
    // Restore stock
    await db.from('ingredients').update({ quantity_on_hand: 100 }).eq('id', ingredientAId);
    await db.from('ingredients').update({ quantity_on_hand: 200 }).eq('id', ingredientBId);
  });

  it('I4: p_allow_negative=true bypasses guard and writes audit_log', async () => {
    await db.from('ingredients').update({ quantity_on_hand: 0 }).in('id', [ingredientAId, ingredientBId]);
    const { error } = await db.rpc('deplete_for_order_item', {
      p_order_item_id: orderItemId,
      p_direction: 1,
      p_allow_negative: true,
    });
    expect(error).toBeNull();
    const { data: auditRows } = await db
      .from('audit_log')
      .select('*')
      .eq('action', 'stock_override')
      .eq('entity_type', 'order_item')
      .eq('entity_id', orderItemId);
    expect((auditRows as unknown[]).length).toBeGreaterThanOrEqual(1);
  });
});
```
